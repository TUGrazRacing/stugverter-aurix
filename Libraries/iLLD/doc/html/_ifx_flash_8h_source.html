430IFX_INLINE uint8 IfxFlash_enterPageMode(uint32 pageAddr)
431{
432 volatile uint32 *addr1 = (volatile uint32 *)(IFXFLASH_CMD_BASE_ADDRESS | 0x5554);
433
434 if ((pageAddr & 0xff000000) == 0xa0000000) // program flash
435 {
436 *addr1 = 0x50;
437 return 0;
438 }
439 else if ((pageAddr & 0xff000000) == 0xaf000000) // data flash
440 {
441 *addr1 = 0x5D;

443 }
