767IFX_INLINE void IfxCpu_enableSegmentSpecificDataAccessCacheability(uint16 segmentNumberMask, boolean enable)
768{
769 uint32 cpu_pmaVal;
770 uint16 checkRestrictionMask;
771 uint32 coreIndex = IfxCpu_getCoreIndex();
772 uint16 wdtPassword = IfxScuWdt_getCpuWatchdogPasswordInline(&MODULE_SCU.WDTCPU[coreIndex]);
773
774 /*resolve the restrictions*/
775 /*In PMA0 Segment-C and Segment[7-CoreID] must have the same value */
776 checkRestrictionMask = ((uint16)1 << (7 - coreIndex)) | ((uint16)1 << 0xC);
777
778 if ((segmentNumberMask & checkRestrictionMask) != 0)
779 {
780 segmentNumberMask |= checkRestrictionMask;
781 }
782
783 cpu_pmaVal = __mfcr(CPU_PMA0); /* Read the CPU_PMA0 */
784
785 cpu_pmaVal = enable ? (cpu_pmaVal | segmentNumberMask) : (cpu_pmaVal & ~segmentNumberMask); /* enable or disable the corresponding bitfield */
786
787 /*The CPU_PMA registers are ENDINIT protected*/
788 IfxScuWdt_clearCpuEndinitInline(&MODULE_SCU.WDTCPU[coreIndex], wdtPassword);
789 /*When changing the value of the CPU_PMAx registers both the instruction and data caches should be invalidated*/
790 /*Write to PMA0 register for selecting the cacheability for data cache*/
791 __dsync(); /* DSYNC instruction should be executed immediately prior to the MTCR*/
792 __mtcr(CPU_PMA0, cpu_pmaVal);
793 __isync(); /* ISYNC instruction executed immediately following MTCR */
794 IfxScuWdt_setCpuEndinitInline(&MODULE_SCU.WDTCPU[coreIndex], wdtPassword);
795}

951IFX_INLINE void IfxCpu_initCSA(uint32 *csaBegin, uint32 *csaEnd)
952{
953 uint32 k;
954 uint32 nxt_cxi_val = 0U;
955 uint32 *prvCsa = csaBegin;
956 uint32 *nxtCsa = csaBegin;
957 uint32 numOfCsa = (((uint32)csaEnd - (uint32)csaBegin) / 64U);
958
959 for (k = 0U; k < numOfCsa; k++)
960 {
961 nxt_cxi_val = ((uint32)nxtCsa & (0XFU << 28U)) >> 12U | ((uint32)nxtCsa & (0XFFFFU << 6U)) >> 6U;
962
963 if (k == 0U)
964 {
965 __mtcr(CPU_FCX, nxt_cxi_val); /* store the new pcxi value to LCX */
966 }
967 else
968 {
969 *prvCsa = nxt_cxi_val; /* Store null pointer in last CSA (= very first time!) */
970 }
971
972 if (k == (numOfCsa - 3U))
973 {
974 __mtcr(CPU_LCX, nxt_cxi_val); /* Last but 2 context save area is pointed in LCX to know if there is CSA depletion */
975 }
976
977 prvCsa = (uint32 *)nxtCsa;
978 nxtCsa += 16U; /* next CSA */
979 }
980
981 *prvCsa = 0U;
982}

1007{
