779IFX_INLINE void IfxMtu_disableMbistShell(IfxMtu_MbistSel mbistSel)
780{
781 volatile uint32 *mtuMemtest = (volatile uint32 *)((uint32)&MTU_MEMTEST0 + 4 * (mbistSel >> 5));
782 uint32 mask = 1 << (mbistSel & 0x1f);
783 *mtuMemtest &= ~mask;
784 /* Set EndInit Watchdog (to prevent Watchdog TO)*/
785 uint16 password = IfxScuWdt_getSafetyWatchdogPassword();
786 IfxScuWdt_setSafetyEndinit(password);
787
788 while (!IfxMtu_isMbistDone(mbistSel))
789 {
790 __nop();
791 }
792
793 /* Clear EndInit */
794 IfxScuWdt_clearSafetyEndinit(password);
795 /*Enable Error Reporting */
796 Ifx_MTU_MC *mc = &MODULE_MTU.MC[mbistSel];
797 mc->ALMSRCS.B.MISCE = 1;
798 mc->ALMSRCS.B.OPENE = 1;
799}

821IFX_INLINE void IfxMtu_enableModule(void)
822{
823 uint8 isEndInitEnabled = 0;
824
825 if (IfxScuWdt_getCpuWatchdogEndInit() == 1U)
826 {
827 /* Clear EndInit */
828 IfxScuWdt_clearCpuEndinit(IfxScuWdt_getCpuWatchdogPassword());
829 isEndInitEnabled = 1;
830 }
831
832 /* MTU clock enable */
833 MTU_CLC.U = 0x0U;
834
835 if (isEndInitEnabled == 1U)
836 {
837 IfxScuWdt_setCpuEndinit(IfxScuWdt_getCpuWatchdogPassword());

842IFX_INLINE uint32 IfxMtu_getMemDoneRegisterIndex(IfxMtu_MbistSel mbistSel)
843{
844 uint32 index;
845
846 if (mbistSel > 63)
847 {
848 index = mbistSel - 64;
849 index = index << 8 | 2;
850 }
851 else if (mbistSel > 31)
852 {
853 index = mbistSel - 32;
854 index = index << 8 | 1;
855 }
856 else
857 {
858 index = mbistSel << 8;
859 }
860
861 return index;

887IFX_INLINE boolean IfxMtu_isMbistDone(IfxMtu_MbistSel mbistSel)
888{
889 volatile uint32 index;
890 volatile uint32 bitIndex;
891 uint32 memDone;
892
893 index = IfxMtu_getMemDoneRegisterIndex(mbistSel);
894 bitIndex = index >> 8;
895 index = index & 0xF;
896 memDone = *((volatile uint32 *)(&MTU_MEMDONE0 + index));
897 return ((memDone) >> bitIndex) & 0x1;
898}
