201IFX_INLINE boolean IfxScuLbist_multiTriggersInline(const IfxScuLbist_MultiTriggerConfig *multiTriggerConfig)
202{
203 boolean returnVal = 0;
204 uint16 password;
205
206 password = IfxScuWdt_getCpuWatchdogPasswordInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()]);
207
208 /* Previous LBIST done?*/
209 if (MODULE_SCU.LBISTCTRL0.B.LBISTDONE == 1U)
210 {
211 /* Does Signature match with the param sets? */
212 if (multiTriggerConfig->parameterSets[MODULE_SCU.RSTCON2.B.USRINFO].signature == MODULE_SCU.LBISTCTRL3.B.SIGNATURE)
213 {
214 /* Is config sets which matched - the Last one in 'param set List'? */
215 if (MODULE_SCU.RSTCON2.B.USRINFO == 0)
216 {
217 returnVal = 1;
218 }
219 else
220 {
221 /* Store the "number of Config sets" to RSTCON2.USRINFO for the first time. */
222 IfxScuWdt_clearCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
223 MODULE_SCU.RSTCON2.B.USRINFO -= 1;
224 IfxScuWdt_setCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
225
226 /* Configure next param set and re-trigger LBIST */
227 IfxScuLbist_retriggerInline(&multiTriggerConfig->parameterSets[MODULE_SCU.RSTCON2.B.USRINFO]);
228 }
229 }
230 else
231 {
232 /* Signature failed */
233 returnVal = 0;
234 }
235 }
236 else
237 {
238 /* Store the "number of Config sets" to RSTCON2.USRINFO for the first time. */
239 IfxScuWdt_clearCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
240 MODULE_SCU.RSTCON2.B.USRINFO = multiTriggerConfig->numOfConfigSets;
241 IfxScuWdt_setCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
242
243 /* Configure LBIST for param set0 and trigger LBIST */
244 IfxScuLbist_triggerInline(&multiTriggerConfig->parameterSets[MODULE_SCU.RSTCON2.B.USRINFO]);
245
246 while (1)
247 {
248 __nop(); /* Wait till warm reset */
249 }
250 }
251
252 return returnVal;
253}

256IFX_INLINE void IfxScuLbist_retriggerInline(const IfxScuLbist_ParameterSet *parameterSet)
257{
258 Ifx_SCU_LBISTCTRL1 LBISTCTRL1;
259 uint16 password = IfxScuWdt_getSafetyWatchdogPasswordInline();
260 IfxScuWdt_clearSafetyEndinitInline(password);
261
262 /* Reset the LBIST before re-trigger and wait till it's done */
263 MODULE_SCU.LBISTCTRL0.B.LBISTRES = 1;
264 IfxScuWdt_setSafetyEndinitInline(password);
265
266 while (MODULE_SCU.LBISTCTRL0.B.LBISTDONE == 1U)
267 {}
268
269 LBISTCTRL1.U = MODULE_SCU.LBISTCTRL1.U;
270 LBISTCTRL1.B.BODY = parameterSet->application;
271 LBISTCTRL1.B.SEED = parameterSet->seed;
272 LBISTCTRL1.B.SPLITSH = parameterSet->splitShiftSel;
273 LBISTCTRL1.B.LBISTFREQU = parameterSet->freq;
274
275 IfxScuWdt_clearSafetyEndinitInline(password);
276 MODULE_SCU.LBISTCTRL1.U = LBISTCTRL1.U;
277
278 MODULE_SCU.LBISTCTRL0.B.PATTERNS = parameterSet->pattern;
279 MODULE_SCU.LBISTCTRL0.B.LBISTREQ = 1;
280
281 while (1)
282 {
283 __nop(); /* Wait till warm reset */
284 }
285}

288IFX_INLINE void IfxScuLbist_triggerInline(const IfxScuLbist_ParameterSet *parameterSet)
289{
290 IfxScuWdt_clearGlobalSafetyEndinitInline(IfxScuWdt_getGlobalSafetyEndinitPasswordInline());
291
292 MODULE_SCU.LBISTCTRL1.U =
293 (parameterSet->application << IFX_SCU_LBISTCTRL1_BODY_OFF) |
294 (parameterSet->seed << IFX_SCU_LBISTCTRL1_SEED_OFF) |
295 (parameterSet->splitShiftSel << IFX_SCU_LBISTCTRL1_SPLITSH_OFF) |
296 (parameterSet->freq << IFX_SCU_LBISTCTRL1_LBISTFREQU_OFF);
297
298 MODULE_SCU.LBISTCTRL0.B.PATTERNS = parameterSet->pattern;
299 MODULE_SCU.LBISTCTRL2.B.LENGTH = parameterSet->scanChainLength;
300 MODULE_SCU.LBISTCTRL0.U |=
301 (IFX_SCU_LBISTCTRL0_LBISTREQRED_MSK << IFX_SCU_LBISTCTRL0_LBISTREQRED_OFF) |
302 (IFX_SCU_LBISTCTRL0_LBISTREQ_MSK << IFX_SCU_LBISTCTRL0_LBISTREQ_OFF);
303}
