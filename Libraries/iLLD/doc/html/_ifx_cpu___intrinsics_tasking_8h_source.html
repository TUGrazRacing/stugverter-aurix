357IFX_INLINE uint32 IfxCpu_calculateCrc32(uint32 *startaddress, uint8 length)
358{
359 uint32 returnvalue;
360 __asm("MOV d0, #0x0"); /* set seed value to 0 */
361 for (;length > 0; length--)
362 {
363 /* calculate the CRC over all data */
364 __asm("LD.W d1,[%0]" : : "a" (startaddress));
365 __asm("CRC32 d0,d0,d1");
366 startaddress++;
367 }
368 __asm("MOV %0,d0" : "=d" (returnvalue)); /* return result of CRC*/
369 return returnvalue;
370}
