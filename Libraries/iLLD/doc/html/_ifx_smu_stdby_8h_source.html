246IFX_INLINE void IfxSmuStdby_clearSmuStdbyMonBistFlags(void)
247{
248 Ifx_PMS_MONBISTCTRL monBistCtrl;
249 uint16 passwd = IfxScuWdt_getSafetyWatchdogPassword();
250
251 /* disable the write-protection for registers */
252 IfxScuWdt_clearSafetyEndinit(passwd);
253
254 monBistCtrl.U = PMS_MONBISTCTRL.U;
255 monBistCtrl.B.TSTCLR = 1;
256 monBistCtrl.B.BITPROT = 1;
257 PMS_MONBISTCTRL.U = monBistCtrl.U;
258
259 /* Restore back the write-protection for registers */
260 IfxScuWdt_setSafetyEndinit(passwd);
261}

313IFX_INLINE uint32 IfxSmuStdby_getSmuStdbyAlarmGroupStatus(uint8 alarmGroup)
314{
315 uint32 alarmGroupStatus = 0;
316
317 if (alarmGroup == 20U)
318 {
319 alarmGroupStatus = PMS_AG_STDBY0.U & 0x0000FFFFU;
320 }
321 else if (alarmGroup == 21U)
322 {
323 alarmGroupStatus = PMS_AG_STDBY1.U;
324 }
325 else
326 {
327 IFX_ASSERT(IFX_VERBOSE_LEVEL_ERROR, FALSE);
328 }
329
