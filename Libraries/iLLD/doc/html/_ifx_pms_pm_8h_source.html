787{
788 uint16 cpuWdtPw = IfxScuWdt_getCpuWatchdogPassword();
789
790 uint8 index = IfxCpu_getCoreIndex();
791 IfxScuWdt_clearCpuEndinit(cpuWdtPw);
792
793 switch (index)
794 {
795 case 0:
796 MODULE_SCU.PMCSR0.B.REQSLP = powerMode;
797 break;
798 case 1:
799 MODULE_SCU.PMCSR1.B.REQSLP = powerMode;
800 break;
801 case 2:
802 MODULE_SCU.PMCSR2.B.REQSLP = powerMode;
803 break;
804 case 3:
805 MODULE_SCU.PMCSR3.B.REQSLP = powerMode;
806 break;
807 }
808
809 IfxScuWdt_setCpuEndinit(cpuWdtPw);
810}
