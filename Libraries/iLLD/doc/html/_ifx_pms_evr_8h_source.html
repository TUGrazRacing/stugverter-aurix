736IFX_INLINE boolean IfxPmsEvr_areInitValuesRight(const IfxPmsEvr_checkRegConfig *const checkConfig)
737{
738 boolean allRight = TRUE;
739 sint32 regCfgCount;
740
741 if (PMS_EVRSTAT.B.EVRC == 1U)
742 {
743 for (regCfgCount = 0; regCfgCount < checkConfig->numOfRegisters; regCfgCount++)
744 {
745 uint32 regVal;
746 const IfxPmsEvr_StepDownRegulatorRegConfig *regConfig = &checkConfig->regConfig[regCfgCount];
747 regVal = (*((volatile uint32 *)regConfig->regAddr));
748 regVal &= regConfig->mask;
749
750 if (regVal != regConfig->value)
751 {
752 allRight = FALSE;
753 }
754 }
755
756 /*Check if EVRSTAT.SDVOK is set*/
757 if (allRight == TRUE)
758 {
759 if (PMS_EVRSTAT.B.SDVOK != 1)
760 {
761 allRight = FALSE;
762 }
763 }
764 }
765
766 return allRight;
767}

926IFX_INLINE boolean IfxPmsEvr_runInitSequence(const IfxPmsEvr_initSequence *const sequence)
927{
928 boolean pmsInitDone = TRUE;
929 uint32 regVal;
930 uint16 endinitSfty_pw = IfxScuWdt_getGlobalSafetyEndinitPasswordInline();
931 const IfxPmsEvr_initSequencePhase *phaseConfig;
932 const IfxPmsEvr_StepDownRegulatorRegConfig *regConfig;
933
934 if (PMS_EVRSTAT.B.EVRC == 1U)
935 {
936 /*Loop through each configured phase*/
937 for (phaseConfig = &sequence->phaseConfig[0];
938 phaseConfig < &sequence->phaseConfig[sequence->numOfPhases];
939 phaseConfig++)
940 {
941 IfxScuWdt_clearGlobalSafetyEndinitInline(endinitSfty_pw);
942
943 /*Loop through each register which is configured in the phase configuration*/
944 for (regConfig = &phaseConfig->regConfig[0];
945 regConfig < &phaseConfig->regConfig[phaseConfig->numOfRegisters];
946 regConfig++)
947 {
948 regVal = (*((volatile uint32 *)regConfig->regAddr));
949 regVal &= ~regConfig->mask;
950 regVal |= regConfig->value;
951
955 (*((volatile uint32 *)regConfig->regAddr)) = regVal; /*Write to the physical register*/
956 }
957
958 PMS_EVRSDCTRL0.B.UP = 1;
959 IfxScuWdt_setGlobalSafetyEndinitInline(endinitSfty_pw);
960
961 /* wait for the UP bit*/
962 sint32 timeoutCount = IFXPMSEVR_CTRL_UP_WAIT_TIMEOUTCOUNT;
963
964 /*Now, wait for the UP bit at EVRSDCTRL0 is reset, which is to indicate the configurations are done*/
965 do
966 {
967 if (--timeoutCount <= 0)
968 {
969 pmsInitDone = FALSE;
970 break;
971 }
972 } while (PMS_EVRSDCTRL0.B.UP == 1);
973
974 IfxPmsEvr_wait(phaseConfig->waitInSecs); // wait for configured time
975 }
976 }
977
978 /*return the status*/
979 return pmsInitDone;
980}

1230IFX_INLINE void IfxPmsEvr_wait(float32 waitInSec)
1231{
1232 uint32 stmCount = (uint32)((float32)33000000UL * waitInSec);
1233 uint32 stmCountBegin = STM0_TIM0.U; /* it is necessary to get this value to have minimum 100uS delay in subsequent CPU start */
1234
1235 while ((uint32)(STM0_TIM0.U - stmCountBegin) < stmCount)
1236 {
1237 /* There is no need to check overflow of the STM timer.
1238 * When counter after overflow subtracted with counter before overflow,
1239 * the subtraction result will be as expected, as long as both are unsigned 32 bits
1240 * eg: stmCountBegin= 0xFFFFFFFE (before overflow)
1241 * stmCountNow = 0x00000002 (before overflow)
1242 * diff= stmCountNow - stmCountBegin = 4 as expected.*/
1243 }
1244}
