667IFX_INLINE void IfxGtm_Atom_setTout(uint32 toutNum, uint32 toutSel)
668{
669 uint32 outselReg = (toutNum >> 3);
670 uint32 shift = (toutNum & 0x7U) * 4;
671 uint32 outsel = toutSel << shift;
672 uint32 mask = 0xFU << shift;
673
674 __ldmst_c(&(MODULE_GTM.TOUTSEL[outselReg].U), mask, outsel);
675}
