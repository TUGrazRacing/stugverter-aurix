473IFX_INLINE void IfxSmu_clearAlarmExecutedStatus(IfxSmu_AlarmExecutionStatus alarmEx)
474{
475 uint16 passwd = IfxScuWdt_getSafetyWatchdogPassword();
476
477 /* disable the write-protection for registers */
478 IfxScuWdt_clearSafetyEndinit(passwd);
479
480 /* Write 1 in AEX bit to clear alarm Executed mechanism */
481 MODULE_SMU.AEXCLR.U = (uint32)((uint32)0x1U << alarmEx);
482
483 /* Restore back the write-protection for registers */
484 IfxScuWdt_setSafetyEndinit(passwd);
485}

576IFX_INLINE void IfxSmu_setPortControlHwEnableAndDir(IfxSmu_PortControlHwEnable portEnable, IfxSmu_PortControlHwDir hwDir)
577{
578 uint16 passwd = IfxScuWdt_getSafetyWatchdogPassword();
579 Ifx_SMU_PCTL pctl;
580 /* disable the write-protection for registers */
581 IfxScuWdt_clearSafetyEndinit(passwd);
582
583 pctl.U = MODULE_SMU.PCTL.U;
584 pctl.B.HWEN = portEnable;
585 pctl.B.HWDIR = hwDir;
586 pctl.B.PCS = 1;
587 MODULE_SMU.PCTL.U = pctl.U;
588
589 /* Restore back the write-protection for registers */
590 IfxScuWdt_setSafetyEndinit(passwd);
