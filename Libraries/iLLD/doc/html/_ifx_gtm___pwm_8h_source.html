623{
624 IfxGtm_Pwm_Channel *channel = &pwm->channels[configIndex];
625
626 /* Phase in ticks = period * (phase in radian / (2 * pi)) */
627 uint32 phaseTicks = (uint32)((float32)(((float32)pwm->periodTicks * requestPhase * IFX_ONE_OVER_TWO_PI) + (float32)0.5f));
628
629 /* Subtract old phase value and add new vale */
630 uint32 dutyTicks = (channel->dutyTicks - channel->phaseTicks) + phaseTicks;
631
632 /* Copy to handle */
633 channel->phaseTicks = phaseTicks;
634 channel->dutyTicks = dutyTicks;
635
636 /* Update shadow register */

667{
668 IfxGtm_Pwm_Channel *channel = &pwm->channels[configIndex];
669
670 /* Duty in ticks = period * (duty in percent / 100%) */
671 uint32 dutyTicks = (uint32)((float32)(((float32)pwm->periodTicks * requestDuty * 0.01f) + (float32)0.5f));
672
673 /* Subtract from period for center aligned */
674 if (pwm->alignment != IfxGtm_Pwm_Alignment_edge)
675 {
676 dutyTicks = pwm->periodTicks - dutyTicks;
677 channel->dutyTicks = dutyTicks;
678 }
679 /* Offset by phase to account for phase shift in sync channel edge aligned (phaseticks = 0 for base channel) */
680 else
681 {
682 dutyTicks += channel->phaseTicks;

684 dutyTicks = (dutyTicks <= pwm->periodTicks) ? dutyTicks : dutyTicks - pwm->periodTicks;
685 }
686
687 /* Update shadow register */
688 *channel->registers.SR1 = dutyTicks;
689}
