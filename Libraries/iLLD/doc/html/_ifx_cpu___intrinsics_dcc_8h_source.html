1535{
1536% reg a,x,m; lab cmp_m, sub_m, done;
1537! "%d4","%d5","%d6","%d0","%d1","%d2"
1538 mul.u %e4,a,x /* d5 = Eh; d4 = El */
1539 mov %d0,%d4 /* e0 = El */
1540 mov %d1, 0

1544 jz %d5,done
1545sub_m:
1546 subx %d4,%d4,m /* e4=e4-m */
1547 subc %d5,%d5,%d1 /* d5=d5-0 */
1548 loopu cmp_m
