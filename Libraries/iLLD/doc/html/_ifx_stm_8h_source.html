700{
701 Ifx_STM_OCS ocs;
702
703 /* Only if OCDS is enabled write into the OCS register */
704 if (MODULE_CBS.OSTATE.B.OEN == 1U)
705 {
706 // remove protection and configure the suspend mode.
707 ocs.B.SUS_P = 1;

721IFX_INLINE void IfxStm_waitTicks(Ifx_STM *stm, uint32 ticks)
722{
723 uint32 beginTime;
724
725 beginTime = IfxStm_getLower(stm);
726
727 /*below code will work because of unsigned 32 bit calculation even at timer wrapping condition
728 * As an example if beginTime = 0xFFFFFFFE and current time = 2 (after overflow), unsigned calculation
729 * 2 - 0xFFFFFFFE will be 4*/
730 while ((IfxStm_getLower(stm) - beginTime) < ticks)
731 {}
732}

749{
