575 IfxGtm_Tom_Ch_ClkSrc clk;
576 IfxGtm_Cmu_Fxclk clkIndex[5] = {
577 IfxGtm_Cmu_Fxclk_0, /*IfxGtm_Tom_Ch_ClkSrc_cmuFxclk0 */
578 IfxGtm_Cmu_Fxclk_1, /*IfxGtm_Tom_Ch_ClkSrc_cmuFxclk1 */

582 };
583
584 clk = IfxGtm_Tom_Ch_getClockSource(tom, channel);
585
586 if (clk == IfxGtm_Tom_Ch_ClkSrc_noClock)
587 {
588 return (float32)0.0f;

592 return IfxGtm_Cmu_getFxClkFrequency(gtm, clkIndex[clk], TRUE);
