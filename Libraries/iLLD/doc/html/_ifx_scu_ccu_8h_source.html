1247IFX_INLINE boolean IfxScuCcu_configureCcuInitialStep(const IfxScuCcu_InitialStepConfig *pllInitStepCfg)
1248{
1249 uint8 initError = 0;
1250 uint16 endinitSfty_pw;
1251
1252 uint32 timeoutCycleCount;
1253 const IfxScuCcu_pllsParameterConfig *pllsParamCfg;
1254 pllsParamCfg = &pllInitStepCfg->pllsParameters;
1255 /*get the watch-dog passwords for usage with this function*/
1256 endinitSfty_pw = IfxScuWdt_getSafetyWatchdogPasswordInline();
1257
1258 /* Reset Safety ENDINIT, SCU_CCUCON registers are protected*/
1259 IfxScuWdt_clearSafetyEndinitInline(endinitSfty_pw);
1260
1261 timeoutCycleCount = IFXSCUCCU_CCUCON_LCK_BIT_TIMEOUT_COUNT;
1262
1263 while (SCU_CCUCON0.B.LCK != 0U)
1264 {
1265 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1266 }
1267
1268 {
1269 Ifx_SCU_CCUCON0 scuCcucon0;
1270 scuCcucon0.U = SCU_CCUCON0.U;
1271 scuCcucon0.B.CLKSEL = 0; /*Select the EVR as fsource0/1/2 for the clock distribution */
1272 scuCcucon0.B.UP = 1; /*Update the ccucon0 register */
1273 SCU_CCUCON0.U = scuCcucon0.U;
1274 }
1275
1276 timeoutCycleCount = IFXSCUCCU_CCUCON_LCK_BIT_TIMEOUT_COUNT;
1277
1278 while (SCU_CCUCON0.B.LCK != 0U)
1279 {
1280 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1281 }
1282
1283 /*Disable SMU Alarms */
1284 {
1285 SMU_KEYS.U = (uint32)0xBCU; /* Enable access to SMU registers */
1286 SMU_AG8CF0.U &= ~IFXSCUCCU_SMUALARM_MASK;
1287 SMU_AG8CF1.U &= ~IFXSCUCCU_SMUALARM_MASK;
1288 SMU_AG8CF2.U &= ~IFXSCUCCU_SMUALARM_MASK;
1289 SMU_KEYS.U = (uint32)0U;
1290 }
1291
1292 /* Power down the both the PLLs before configuring registers*/
1293 /* Both the PLLs are powered down to be sure for asynchronous PLL registers update cause no glitches */
1294 SCU_SYSPLLCON0.B.PLLPWD = 0;
1295 SCU_PERPLLCON0.B.PLLPWD = 0;
1296
1297 timeoutCycleCount = IFXSCUCCU_SYSPLLSTAT_PWDSTAT_TIMEOUT_COUNT;
1298
1299 while ((SCU_SYSPLLSTAT.B.PWDSTAT == 0) || (SCU_PERPLLSTAT.B.PWDSTAT == 0))
1300 {
1301 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1302 }
1303
1304 /* Now configure the oscillator, required oscillator mode is external crystal */
1305
1306 if ((pllsParamCfg->pllInputClockSelection == IfxScuCcu_PllInputClockSelection_fOsc0) || (pllsParamCfg->pllInputClockSelection == IfxScuCcu_PllInputClockSelection_fSysclk))
1307
1308 {
1309 Ifx_SCU_OSCCON scuOsccon;
1310 scuOsccon.U = SCU_OSCCON.U;
1311#if (IFX_CFG_SCU_EXT_CLOCK == 1)
1312 scuOsccon.B.MODE = 2U; /* external clock selected */
1313#else
1314 scuOsccon.B.MODE = 0U;
1315#endif
1316
1317 scuOsccon.B.OSCVAL = (uint32)(pllsParamCfg->xtalFrequency / 1000000U) - 15; /*XTAL range 16MHz to 40MHz*/
1318 /*Note: Oscillator Watchdog Reset (OSCRES) is not required when the PLLs are powered down*/
1319 SCU_OSCCON.U = scuOsccon.U;
1320 }
1321
1322 /* Configure the initial steps for the system PLL*/
1323 {
1324 Ifx_SCU_SYSPLLCON0 sysPllCon0;
1325 sysPllCon0.U = SCU_SYSPLLCON0.U;
1326 sysPllCon0.B.PDIV = pllsParamCfg->sysPllConfig.pDivider;
1327 sysPllCon0.B.NDIV = pllsParamCfg->sysPllConfig.nDivider;
1328 sysPllCon0.B.INSEL = pllsParamCfg->pllInputClockSelection;
1329 SCU_SYSPLLCON0.U = sysPllCon0.U;
1330 }
1331
1332 /* Configure the initial steps for the peripheral PLL*/
1333 {
1334 Ifx_SCU_PERPLLCON0 scuPerPllCon0;
1335 scuPerPllCon0.U = SCU_PERPLLCON0.U;
1336 scuPerPllCon0.B.DIVBY = pllsParamCfg->perPllConfig.k3DividerBypass;
1337 scuPerPllCon0.B.PDIV = pllsParamCfg->perPllConfig.pDivider;
1338 scuPerPllCon0.B.NDIV = pllsParamCfg->perPllConfig.nDivider;
1339 SCU_PERPLLCON0.U = scuPerPllCon0.U;
1340 }
1341
1342 /* Power up the both the PLLs before configuring registers*/
1343 SCU_SYSPLLCON0.B.PLLPWD = 1;
1344 SCU_PERPLLCON0.B.PLLPWD = 1;
1345
1346 timeoutCycleCount = IFXSCUCCU_SYSPLLSTAT_PWDSTAT_TIMEOUT_COUNT;
1347
1348 while ((SCU_SYSPLLSTAT.B.PWDSTAT == 1) || (SCU_PERPLLSTAT.B.PWDSTAT == 1))
1349 {
1350 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1351 }
1352
1353 timeoutCycleCount = IFXSCUCCU_PLL_KRDY_TIMEOUT_COUNT;
1354
1355 while ((SCU_SYSPLLSTAT.B.K2RDY == 0U) ||
1356 (SCU_PERPLLSTAT.B.K2RDY == 0U) ||
1357 (SCU_PERPLLSTAT.B.K3RDY == 0U))
1358 {
1359 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1360 }
1361
1362 SCU_SYSPLLCON1.B.K2DIV = pllsParamCfg->sysPllConfig.k2Divider;
1363 {
1364 Ifx_SCU_PERPLLCON1 scuPerPllCon1;
1365 scuPerPllCon1.U = SCU_PERPLLCON1.U;
1366 scuPerPllCon1.B.K2DIV = pllsParamCfg->perPllConfig.k2Divider;
1367 scuPerPllCon1.B.K3DIV = pllsParamCfg->perPllConfig.k3Divider;
1368 SCU_PERPLLCON1.U = scuPerPllCon1.U;
1369 }
1370
1371 timeoutCycleCount = IFXSCUCCU_PLL_KRDY_TIMEOUT_COUNT;
1372
1373 while ((SCU_SYSPLLSTAT.B.K2RDY == 0U) ||
1374 (SCU_PERPLLSTAT.B.K2RDY == 0U) ||
1375 (SCU_PERPLLSTAT.B.K3RDY == 0U))
1376 {
1377 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1378 }
1379
1380 /*Check if the OSC frequencies are in the limit*/
1381 timeoutCycleCount = IFXSCUCCU_OSCCON_PLLLV_OR_HV_TIMEOUT_COUNT;
1382
1383 while ((SCU_OSCCON.B.PLLLV == 0) && (SCU_OSCCON.B.PLLHV == 0))
1384 {
1385 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1386 }
1387
1388 /* Now start PLL locking for latest set values*/
1389 {
1390 SCU_SYSPLLCON0.B.RESLD = 1; /* Restart Sys PLL lock detection */
1391 SCU_PERPLLCON0.B.RESLD = 1; /* Restart Per PLL lock detection */
1392
1393 timeoutCycleCount = IFXSCUCCU_PLL_LOCK_TIMEOUT_COUNT;
1394
1395 while ((SCU_SYSPLLSTAT.B.LOCK == 0) || (SCU_PERPLLSTAT.B.LOCK == 0))
1396 {
1397 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1398 }
1399 }
1400 /*Enable SMU Alarms */
1401 {
1402 SMU_KEYS.U = (uint32)0xBCU; /* Enable access to SMU registers */
1403 SMU_CMD.U = (uint32)0x00000005;
1404 SMU_AG8.U = IFXSCUCCU_SMUALARM_MASK; /* Clear SMU Alarms*/
1405 SMU_KEYS.U = (uint32)0U;
1406 }
1407 {
1408 Ifx_SCU_CCUCON0 scu_ccucon0;
1409 scu_ccucon0.U = SCU_CCUCON0.U;
1410 scu_ccucon0.B.CLKSEL = 1; /*Select the PLLs for fsource0/1/2 */
1411 scu_ccucon0.B.UP = 1; /*Update the ccucon0 register */
1412
1413 timeoutCycleCount = IFXSCUCCU_CCUCON_LCK_BIT_TIMEOUT_COUNT;
1414
1415 while (SCU_CCUCON0.B.LCK != 0U)
1416 {
1417 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1418 }
1419
1420 SCU_CCUCON0.U = scu_ccucon0.U;
1421
1422 timeoutCycleCount = IFXSCUCCU_CCUCON_LCK_BIT_TIMEOUT_COUNT;
1423
1424 while (SCU_CCUCON0.B.LCK != 0U)
1425 {
1426 IFXSCUCCU_LOOP_TIMEOUT_CHECK(timeoutCycleCount, initError);
1427 }
1428 }
1429
1430 IfxScuWdt_setSafetyEndinitInline(endinitSfty_pw);
1431 return (boolean)initError;
1432}

1886IFX_INLINE void IfxScuCcu_switchToNormalMode(const uint32 ccucon0)
1887{
1888 uint16 endinitSfty_pw;
1889
1890 endinitSfty_pw = IfxScuWdt_getSafetyWatchdogPassword();
1891
1892 /* clear safety ENDINIT prior to write to SCU_CCUCON0 */
1893 IfxScuWdt_clearSafetyEndinit(endinitSfty_pw);
1894
1895 /* wait for unlock status - safety watchdog ensures timeout */
1896 while (SCU_CCUCON0.B.LCK != 0U)
1897 {} /*Wait till ccucon0 lock is set, No "timeout" required, ENDINIT watch-dog is active! */
1898
1899 /* Intermediate step : SPBDIV = 12 and SRIDIV = 3
1900 * NOTE: This intermediate step in division ratio is done to avoid a sudden jump in clock frequency
1901 * while jumping back from Low Power to Normal mode. The division ratios used in this step is hard
1902 * coded into this function after proving on HW.
1903 */
1904 Ifx_SCU_CCUCON0 scu_ccucon0;
1905 scu_ccucon0.U = SCU_CCUCON0.U; /* read ccucon0*/
1906 scu_ccucon0.B.SRIDIV = 3; /* SRIDIV = 1:3*/
1907 scu_ccucon0.B.SPBDIV = 12; /* SPBDIV = 1:12 */
1908 scu_ccucon0.B.LPDIV = 0; /* Exit Low Power Mode */
1909 scu_ccucon0.B.UP = 1;
1910 SCU_CCUCON0.U = scu_ccucon0.U; /* write back */
1911 IfxScuWdt_setSafetyEndinit(endinitSfty_pw); /* set Safety ENDINIT */
1912
1913 /* wait */
1914 IfxScuCcu_wait(IFXSCUCCU_LPTONORMAL_WAITTIME); /* wait */
1915
1916 /* Now back to original settings for normal mode */
1917 /* clear safety ENDINIT prior to write to SCU_CCUCON0 */
1918 IfxScuWdt_clearSafetyEndinit(endinitSfty_pw);
1919
1920 /* wait for unlock status - safety watchdog ensures timeout */
1921 while (SCU_CCUCON0.B.LCK != 0U)
1922 {} /*Wait till ccucon0 lock is set, No "timeout" required, ENDINIT watch-dog is active! */
1923
1924 scu_ccucon0.U = SCU_CCUCON0.U; /* read ccucon0*/
1925 scu_ccucon0.U = ccucon0; /* original setting */
1926 scu_ccucon0.B.UP = 1; /* update */
1927 SCU_CCUCON0.U = scu_ccucon0.U; /* write back */
1928 IfxScuWdt_setSafetyEndinit(endinitSfty_pw); /* set Safety ENDINIT */
1929}
