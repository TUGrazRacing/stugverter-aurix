550IFX_INLINE void IfxScuWdt_clearCpuEndinitInline(Ifx_SCU_WDTCPU *watchdog, uint16 password)
551{
552 if (watchdog->CON0.B.LCK)
553 {
554 /* see Table 1 (Password Access Bit Pattern Requirements) */
555 watchdog->CON0.U = (1 << IFX_SCU_WDTCPU_CON0_ENDINIT_OFF) |
556 (0 << IFX_SCU_WDTCPU_CON0_LCK_OFF) |
557 (password << IFX_SCU_WDTCPU_CON0_PW_OFF) |
558 (watchdog->CON0.B.REL << IFX_SCU_WDTCPU_CON0_REL_OFF);
559 }
560
561 /* Clear ENDINT and set LCK bit in Config_0 register */
562 watchdog->CON0.U = (0 << IFX_SCU_WDTCPU_CON0_ENDINIT_OFF) |
563 (1 << IFX_SCU_WDTCPU_CON0_LCK_OFF) |
564 (password << IFX_SCU_WDTCPU_CON0_PW_OFF) |
565 (watchdog->CON0.B.REL << IFX_SCU_WDTCPU_CON0_REL_OFF);
566
567 /* read back ENDINIT and wait until it has been cleared */
568 while (watchdog->CON0.B.ENDINIT == 1)
569 {}
570}

608 uint16 password;
609
610 /* Read Password from CON0 register
