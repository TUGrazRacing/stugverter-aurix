188IFX_INLINE void IfxScuRcu_clearColdResetStatus(void)
189{
190 uint16 password;
191
192 password = IfxScuWdt_getCpuWatchdogPasswordInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()]);
193 IfxScuWdt_clearCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
194 /* Clear Cold Reset status */
195 MODULE_SCU.RSTCON2.B.CLRC = 1U;
196 IfxScuWdt_setCpuEndinitInline(&MODULE_SCU.WDTCPU[IfxCpu_getCoreIndex()], password);
197}

200IFX_INLINE void IfxScuRcu_configureResetRequestTrigger(IfxScuRcu_Trigger trigger, IfxScuRcu_ResetType resetType)
201{
202 Ifx_SCU_RSTCON Rstcon;
203 uint8 resetConfig = 2; /* Default - Application reset */
204 uint16 password;
205 password = IfxScuWdt_getSafetyWatchdogPassword();
206 Rstcon.U = MODULE_SCU.RSTCON.U;
207
208 if (resetType == IfxScuRcu_ResetType_system)
209 {
210 resetConfig = 1;
211 }
212
213 switch (trigger)
214 {
215 case IfxScuRcu_Trigger_esr0:
216 Rstcon.B.ESR0 = resetConfig;
217 break;
218 case IfxScuRcu_Trigger_esr1:
219 Rstcon.B.ESR1 = resetConfig;
220 break;
221 case IfxScuRcu_Trigger_smu:
222 Rstcon.B.SMU = resetConfig;
223 break;
224 case IfxScuRcu_Trigger_sw:
225 Rstcon.B.SW = resetConfig;
226 break;
227 case IfxScuRcu_Trigger_stm0:
228 Rstcon.B.STM0 = resetConfig;
229 break;
230 case IfxScuRcu_Trigger_stm1:
231 Rstcon.B.STM1 = resetConfig;
232 break;
233 case IfxScuRcu_Trigger_stm2:
234 Rstcon.B.STM2 = resetConfig;
235 break;
236 case IfxScuRcu_Trigger_stm3:
237 Rstcon.B.STM3 = resetConfig;
238 break;
239 default:
240 break;
241 }
242
243 IfxScuWdt_clearSafetyEndinitInline(password);
244
245 MODULE_SCU.RSTCON.U = Rstcon.U;
246 IfxScuWdt_setSafetyEndinit(password);
247}

250IFX_INLINE boolean IfxScuRcu_isApplicationReset(uint32 rawResetStatus)
251{
252 boolean returnVal;
253 uint32 tempValue;
254
255 if ((rawResetStatus & IFXSCURCU_POWERONRESET_MASK) > 0)
256 {
257 returnVal = 0;
258 }
259 else if ((rawResetStatus & IFXSCURCU_APPLICATIONRESET_MASK) > 0) // CB3 TP HSMA ??
260 {
261 // get the bit which is set at RSTSTAT (assumption: RSTSTAT register was cleared before and only one bit can be set)
262 tempValue = rawResetStatus & IFXSCURCU_APPLICATIONRESET_MASK;
263 tempValue = (SCU_RSTCON.U >> ((31U - __clz(tempValue)) * 2)) & 3U;
264
265 if (tempValue == 2)
266 {
267 returnVal = 1;
268 }
269 else
270 {
271 returnVal = 0;
272 }
273 }
274 else if ((rawResetStatus & (IFX_SCU_RSTSTAT_CB3_MSK << IFX_SCU_RSTSTAT_CB3_OFF)))
275 {
276 // CB3 reset is not configurable and is always application reset
277 returnVal = 1;
278 }
279 else if (CPU0_KRST0.B.RSTSTAT != 0) /*CPU0 KRST status */
280 {
281 returnVal = 1;
282 }
283 else
284 {
285 returnVal = 0;
286 }
287
288 return returnVal;
289}
